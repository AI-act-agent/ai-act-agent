📊 복합 스코어링 공식 (Composite Scoring Formula)
// 최종 점수 산출 공식
FinalScore = D_score(Q1) × 0.15 + R_score(Q2) × 0.15 + A_score(Q3) × 0.35 + H_score(Q4) × 0.35
// 비가역성 패널티
AdjustedScore = FinalScore + IRR_bonus(비가역=true ? 2 : 0)
// 규모 승수 (영향 받는 사용자 수 기준)
WeightedScore = AdjustedScore × Scale_multiplier(1만↑=1.2, 100만↑=1.4)
